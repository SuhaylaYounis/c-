#include <iostream> // program to update every array element by multipliaction of next and prevoius values
using namespace std;
void update(int arr[], int n)
{
    if (n <= 1)
        return;
    int prev = arr[0];
    arr[0] = arr[0] * arr[1]; // the updated vallue of the first element
    for (int i = 1; i < n - 1; i++) //the updated value of elements from 1 to 4
    {
        int curr = arr[i];
        arr[i] = prev * arr[i + 1];
        prev = curr;
    }
    arr[n - 1] = prev * arr[n-1];   //the updated element of the last element
}
int main()
{
    int arr[] = { 1,2,3,4,5,6 };
    int n = sizeof (arr) / sizeof (arr[0]);
    update(arr, n);
    for (int i = 0; i < n; i++)
    
        cout << arr[i] << "  ";
        return 0;
    
}
